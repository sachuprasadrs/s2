#!/bin/bash

# Function to check if a number is prime
is_prime() {
  num=$1
  if [ $num -le 1 ]; then
    return 1  # Not prime
  fi
  for ((i = 2; i * i <= num; i++)); do
    if ((num % i == 0)); then
      return 1  # Not prime
    fi
  done
  return 0  # Prime
}

# Ask user for the lower and upper limits
echo "Enter the lower limit:"
read lower_limit
echo "Enter the upper limit:"
read upper_limit

# Print prime numbers between the two limits
for ((i = lower_limit; i <= upper_limit; i++)); do
  is_prime $i
  if [ $? -eq 0 ]; then
    echo $i
  fi
done
